STEP 9 — AJUSTAR RESOLVER (FRONTEND)

Arquivo:
client/src/utils/priceResolver.ts

Atualizar função:

ts
export function resolvePrice({
  basePrice,
  subCategoryPrice,
  contractPrice,
  adminFee,
  useNewPricing,
  pricingMode
}: any) {
  let price = 0;

  if (pricingMode === "base") {
    price = basePrice ?? 0;
  } else {
    price = contractPrice ?? subCategoryPrice ?? 0;
  }

  if (useNewPricing && adminFee) {
    price = price * (1 + adminFee / 100);
  }

  return Number(price.toFixed(2));
}
