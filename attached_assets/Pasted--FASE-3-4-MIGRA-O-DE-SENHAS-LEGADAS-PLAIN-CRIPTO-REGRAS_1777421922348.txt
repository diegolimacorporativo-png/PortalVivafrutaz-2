Adicionar:

ts import { isEncrypted, encrypt } from '@/utils/crypto';  export async function migrateLegacyCertificates(): Promise<{   total: number;   migrated: number; }> {   const rows = await db.select().from(companyCertificates);    let migrated = 0;    for (const row of rows) {     if (!row.certPassword) continue;      // já está criptografado → pula     if (isEncrypted(row.certPassword)) continue;      const encrypted = encrypt(row.certPassword);      await db       .update(companyCertificates)       .set({         certPassword: encrypted,         updatedAt: new Date(),       })       .where(eq(companyCertificates.id, row.id));      migrated++;   }    return {     total: rows.length,     migrated,   }; }
