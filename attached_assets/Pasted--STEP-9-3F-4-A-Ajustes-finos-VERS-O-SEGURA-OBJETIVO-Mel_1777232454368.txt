# 🗄️ FASE 3 — PRUNE (COM LOG MELHORADO)

ts import { lt } from "drizzle-orm";  export async function pruneOldAlertLogs(days = 90) {   try {     const cutoff = new Date();     cutoff.setDate(cutoff.getDate() - days);      const result = await db       .delete(cronAlertLogs)       .where(lt(cronAlertLogs.createdAt, cutoff));      console.log("[ALERT_LOGS_PRUNED]", {       days,       deleted: result.rowCount ?? "unknown",     });   } catch (err) {     console.error("[ALERT_PRUNE_ERROR]", err);   } }
