# 🧱 FASE 2 — ENDPOINT ANALYTICS (CORRIGIDO)

## Arquivo: server/routes/routes.ts

ts id="adj1" import { sql } from "drizzle-orm"; import { cronAlertLogs } from "@shared/schema";  app.get(   "/api/cron/alerts/analytics",   requireAuthCore,   requireRole(["MASTER", "ADMIN", "DIRECTOR"]),   async (req, res) => {     try {       const days = Math.min(90, Math.max(1, Number(req.query.days ?? 7)));        const cutoff = new Date();       cutoff.setDate(cutoff.getDate() - days);        // 🔹 Totais (Postgres retorna string → converter depois)       const totals = await db.execute(sql`         SELECT           COUNT(*) as total,           COUNT(*) FILTER (WHERE rate_limited = true) as rate_limited,           COUNT(*) FILTER (WHERE rate_limited = false) as sent         FROM cron_alert_logs         WHERE created_at >= ${cutoff}       `);        // 🔹 Por severidade       const bySeverity = await db.execute(sql`         SELECT severity, COUNT(*) as count         FROM cron_alert_logs         WHERE created_at >= ${cutoff}         GROUP BY severity       `);        // 🔹 Por canal (JSONB seguro)       const byChannel = await db.execute(sql`         SELECT           COALESCE(r->>'channel', 'unknown') as channel,           COUNT(*) as count         FROM cron_alert_logs,         jsonb_array_elements(COALESCE(results, '[]'::jsonb)) as r         WHERE created_at >= ${cutoff}         GROUP BY channel       `);        // 🔹 Top títulos       const topTitles = await db.execute(sql`         SELECT title, COUNT(*) as count         FROM cron_alert_logs         WHERE created_at >= ${cutoff}         GROUP BY title         ORDER BY count DESC         LIMIT 5       `);        // 🔧 Normalização segura       const totalsRaw = totals.rows[0] ?? {};        const response = {         windowDays: days,          totals: {           total: Number(totalsRaw.total ?? 0),           rate_limited: Number(totalsRaw.rate_limited ?? 0),           sent: Number(totalsRaw.sent ?? 0),         },          bySeverity: bySeverity.rows.map((r) => ({           severity: r.severity,           count: Number(r.count),         })),          byChannel: byChannel.rows.map((r) => ({           channel: r.channel ?? "unknown",           count: Number(r.count),         })),          topTitles: topTitles.rows.map((r) => ({           title: r.title,           count: Number(r.count),         })),       };        return res.json(response);     } catch (err) {       console.error("[ALERT_ANALYTICS_ERROR]", err);       return res.status(500).json({ ok: false });     }   } );
