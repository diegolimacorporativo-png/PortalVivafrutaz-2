## STEP 2 — NÃO TOCAR NO BLOCO EXISTENTE

  await tx.execute(sql    UPDATE deliveries     SET status = 'em_rota'     WHERE order_id = ${orderId}       AND status = 'pendente'  );

## STEP 3 — ADICIONAR BLOCO PARA DELIVERED

Logo abaixo do SHIPPED, inserir:

if (to === OrderStatus.DELIVERED) {
  try {
    const result = await tx.execute(sql      UPDATE deliveries       SET status = 'entregue',           delivered_at = NOW(),           updated_at = NOW()       WHERE order_id = ${orderId}         AND status != 'entregue'       RETURNING id    );

    const rows = Array.isArray(result)
      ? result
      : (result as any)?.rows ?? [];

    if (rows.length > 0) {
      try {
        await createLog({
          action: "DELIVERY_COMPLETED",
          description: `Entrega concluída para pedido ${orderCode}`,
          userId: user?.id ?? null,
          userRole: user?.role ?? null,
        });
      } catch () {}
    }
  } catch () {
    // nunca quebrar o fluxo principal
  }
}
