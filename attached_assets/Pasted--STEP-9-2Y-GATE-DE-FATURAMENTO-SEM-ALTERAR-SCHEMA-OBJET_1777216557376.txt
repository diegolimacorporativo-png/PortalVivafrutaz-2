### STEP 2 — IMPLEMENTAÇÃO

ts import { db } from "@/db"; import { sql } from "drizzle-orm";  export async function canEmitNFe(orderId: number) {   const result = await db.execute(sql`     SELECT        o.id,       o.status,       o.fiscal_status,       o.delivery_date,       c.contract_start_date,       c.contract_end_date     FROM orders o     JOIN companies c ON c.id = o.company_id     WHERE o.id = ${orderId}   `);    const order = result.rows?.[0];    if (!order) {     return { allowed: false, reason: "Pedido não encontrado" };   }    const now = new Date();    if (order.status === "CANCELLED") {     return { allowed: false, reason: "Pedido cancelado" };   }    if (order.fiscal_status !== "nota_liberada") {     return { allowed: false, reason: "Pedido ainda não liberado para faturamento" };   }    if (order.delivery_date && new Date(order.delivery_date) > now) {     return { allowed: false, reason: "Entrega ainda não realizada" };   }    if (order.contract_end_date && new Date(order.contract_end_date) < now) {     return { allowed: false, reason: "Contrato da empresa expirado" };   }    return { allowed: true }; }
