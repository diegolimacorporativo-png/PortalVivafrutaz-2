ts async getNfeResumoPorStatus(): Promise<   { status: string; total: number }[] > {   const rows = await db     .select({       status: nfeEmissoes.status,       total: sql<number>`count(*)::int`,     })     .from(nfeEmissoes)     .innerJoin(orders, eq(orders.id, nfeEmissoes.orderId))     .where(eq(orders.companyId, requireTenantId()))     .groupBy(nfeEmissoes.status);    return rows.map((r) => ({     status: r.status ??
