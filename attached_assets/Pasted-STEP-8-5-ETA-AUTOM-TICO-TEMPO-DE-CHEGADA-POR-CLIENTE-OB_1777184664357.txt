## STEP 1 — FUNÇÃO DE DISTÂNCIA

Reutilizar (se já existir):

ts function distanceKm(a, b) {   const R = 6371;   const dLat = (b.lat - a.lat) * Math.PI / 180;   const dLon = (b.lng - a.lng) * Math.PI / 180;    const lat1 = a.lat * Math.PI / 180;   const lat2 = b.lat * Math.PI / 180;    const x = dLon * Math.cos((lat1 + lat2) / 2);   const y = dLat;    return Math.sqrt(x * x + y * y) * R; }

### IMPLEMENTAÇÃO

ts const AVG_SPEED_KMH = 40; // ajustar conforme cidade  export function calculateETA(stops, driverPosition) {   if (!stops.length) return [];    let current = driverPosition || stops[0];   let cumulativeTime = 0;    return stops.map((stop, index) => {     const dist = distanceKm(current, stop);     const timeHours = dist / AVG_SPEED_KMH;     const timeMinutes = timeHours * 60;      cumulativeTime += timeMinutes;      current = stop;      return {       ...stop,       etaMinutes: Math.round(cumulativeTime),       etaAt: new Date(Date.now() + cumulativeTime * 60000)     };   }); }
