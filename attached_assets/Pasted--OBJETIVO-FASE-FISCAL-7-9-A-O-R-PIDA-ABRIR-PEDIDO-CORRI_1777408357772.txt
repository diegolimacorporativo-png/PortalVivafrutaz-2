Alterar query de motivos:

ts async getNfeMotivosRejeicao(): Promise<   {     status: string;     cStat: string;     xMotivo: string;     total: number;     orderId: number;   }[] > {   const rows = await db     .select({       status: nfeEmissoes.status,       orderId: nfeEmissoes.orderId,       cStat: sql<string | null>`substring(         ${nfeEmissoes.xmlAutorizado} FROM '<cStat>(\\d+)</cStat>'       )`,       xMotivo: sql<string | null>`substring(         ${nfeEmissoes.xmlAutorizado} FROM '<xMotivo>([^<]+)</xMotivo>'       )`,       total: sql<number>`count(*)::int`,     })     .from(nfeEmissoes)     .innerJoin(orders, eq(orders.id, nfeEmissoes.orderId))     .where(eq(orders.companyId, requireTenantId()))     .groupBy(sql`1,2,3,4`);      return rows.map((r) => ({     status: r.status ?? "N/D",     orderId: r.orderId,     cStat: r.cStat ?? "N/D",     xMotivo: r.xMotivo ?? "Sem descrição",     total: Number(r.total) || 0,   })); }
