# 🛠️ ETAPA 1 — RATE LIMIT GLOBAL (SIMPLES E SEGURO)

📍 Arquivo (criar se não existir):

server/core/http/rateLimit.ts

---

Adicionar:

ts import type { Request, Response, NextFunction } from "express";  const memory = new Map<string, { count: number; time: number }>();  const WINDOW = 60_000; // 1 min const LIMIT = 60; // 60 req/min por IP  export function simpleRateLimit(req: Request, res: Response, next: NextFunction) {   const ip = req.ip || "unknown";   const now = Date.now();    const entry = memory.get(ip);    if (!entry) {     memory.set(ip, { count: 1, time: now });     return next();   }    if (now - entry.time > WINDOW) {     memory.set(ip, { count: 1, time: now });     return next();   }    entry.count++;    if (entry.count > LIMIT) {     return res.status(429).json({       success: false,       error: { message: "Too many requests", code: "RATE_LIMIT" }     });   }    next(); } 

---

📍 Aplicar em:

server/routes/routes.ts

---

ANTES de registrar rotas:

ts app.use(simpleRateLimit); 

---

⚠️ NÃO aplicar em rotas internas de cron
