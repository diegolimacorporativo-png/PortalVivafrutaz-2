# 🛠️ ETAPA 1 — CENTRALIZAR LOGS DE SEGURANÇA

Criar arquivo:

server/core/security/securityLogger.ts

---

ts type SecurityEvent = {   type: string;   ip?: string;   userId?: number;   path?: string;   timestamp: number; };  const events: SecurityEvent[] = [];  export function logSecurityEvent(event: SecurityEvent) {   events.push({     ...event,     timestamp: Date.now(),   });    // manter só últimos 1000 eventos   if (events.length > 1000) {     events.shift();   } }  export function getSecurityEvents() {   return events; }

# 🛠️ ETAPA 2 — INTEGRAR COM LOGS EXISTENTES

- RATE_LIMITED
- HIGH_RISK_ACTION
- CRITICAL_ACTION

ts logSecurityEvent({   type: 'RATE_LIMIT',   ip: req.ip,   path: req.originalUrl, });

# 🛠️ ETAPA 4 — AGRUPAMENTO SIMPLES

ts export function getTopIPs() {   const map: Record<string, number> = {};    for (const e of events) {     if (!e.ip) continue;     map[e.ip] = (map[e.ip] || 0) + 1;   }    return Object.entries(map)     .sort((a, b) => b[1] - a[1])     .slice(0, 10); }
