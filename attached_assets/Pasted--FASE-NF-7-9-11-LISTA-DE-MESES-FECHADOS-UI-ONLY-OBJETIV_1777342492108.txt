  >
    <div className="text-xs font-semibold text-red-700 mb-2 flex items-center gap-1">

        const label = ${String(c.month).padStart(2, '0')}/${c.year};
        return (
          <span
            key={${c.year}-${c.month}}
            className="px-2 py-1 text-[10px] font-semibold rounded-md border border-red-300 text-red-700 bg-white"
            data-testid={badge-closed-${c.year}-${c.month}}
            title={Fechado em ${new Date(c.closedAt).toLocaleDateString()}}
          >
            🔒 {label}
          </span>
        );
      })}
    </div>
  </div>
