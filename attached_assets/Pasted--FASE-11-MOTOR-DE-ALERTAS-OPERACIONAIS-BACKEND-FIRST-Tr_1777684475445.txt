## 🛠️ ETAPA 1 — CRIAR ALERT ENGINE (NOVO ARQUIVO)

📄 server/core/security/alertEngine.ts

ts type AlertLevel = "LOW" | "MEDIUM" | "HIGH" | "CRITICAL";  interface AlertEvent {   id: string;   type: string;   level: AlertLevel;   message: string;   timestamp: number;   count: number; }  const alertBuffer: Map<string, AlertEvent> = new Map();  const ALERT_WINDOW_MS = 60_000;  function classify(level: string): AlertLevel {   if (level.includes("CRITICAL")) return "CRITICAL";   if (level.includes("FINANCIAL")) return "CRITICAL";   if (level.includes("AFTER_CREATE")) return "HIGH";   if (level.includes("NFE_")) return "HIGH";   if (level.includes("SECURITY")) return "HIGH";   return "MEDIUM"; }  export function pushAlert(type: string, message: string) {   const now = Date.now();    const existing = alertBuffer.get(type);    if (existing && now - existing.timestamp < ALERT_WINDOW_MS) {     existing.count++;     existing.timestamp = now;     return;   }    alertBuffer.set(type, {     id: `${type}-${now}`,     type,     level: classify(type),     message,     timestamp: now,     count: 1,   }); }  export function getAlerts(): AlertEvent[] {   return Array.from(alertBuffer.values()).sort(     (a, b) => b.timestamp - a.timestamp   ); }
