### IMPLEMENTAÇÃO

ts id="track1" export async function getRouteTracking(req, res) {   try {     const { routeId } = req.params;      // 1. rota + motorista     const routeResult = await db.execute(sql`       SELECT id, driver_id       FROM logistics_routes       WHERE id = ${routeId}     `);      const route = routeResult.rows?.[0];     if (!route) return res.status(404).json({ error: "Route not found" });      // 2. stops (rota ordenada)     const stopsResult = await db.execute(sql`       SELECT delivery_id, sequence, latitude, longitude       FROM route_stops       WHERE route_id = ${routeId}       ORDER BY sequence ASC     `);      // 3. última posição do motorista     const gpsResult = await db.execute(sql`       SELECT latitude, longitude, created_at       FROM driver_gps_positions       WHERE driver_id = ${route.driver_id}       ORDER BY created_at DESC       LIMIT 1     `);      return res.json({       route,       stops: stopsResult.rows ?? [],       driver: gpsResult.rows?.[0] ?? null     });    } catch (err) {     console.error("TRACKING_ERROR", err);     res.status(500).json({ error: "tracking_failed" });   } }
