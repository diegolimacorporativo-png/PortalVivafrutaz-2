ts import { getSecurityEvents } from './securityLogger';  export function analyzeSecurity() {   const events = getSecurityEvents();    const ipStats: Record<string, {     total: number;     rateLimit: number;     critical: number;   }> = {};    for (const e of events) {     if (!e.ip) continue;      if (!ipStats[e.ip]) {       ipStats[e.ip] = { total: 0, rateLimit: 0, critical: 0 };     }      ipStats[e.ip].total++;      if (e.type === 'RATE_LIMITED') {       ipStats[e.ip].rateLimit++;     }      if (e.type === 'CRITICAL_ACTION') {       ipStats[e.ip].critical++;     }   }    return Object.entries(ipStats).map(([ip, stats]) => {     let risk = 'LOW';      if (stats.rateLimit > 5) risk = 'MEDIUM';     if (stats.rateLimit > 10) risk = 'HIGH';     if (stats.critical > 3) risk = 'CRITICAL';      return {       ip,       ...stats,       risk     };   }); }

ts export function detectSpike() {   const events = getSecurityEvents();    const lastMinute = Date.now() - 60000;    const recent = events.filter(e => e.timestamp > lastMinute);    return {     lastMinuteEvents: recent.length,     spike: recent.length > 50   }; }
