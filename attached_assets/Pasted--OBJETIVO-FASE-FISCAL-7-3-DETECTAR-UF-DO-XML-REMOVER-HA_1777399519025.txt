## 🔴 1. CRIAR FUNÇÃO DE EXTRAÇÃO

📍 server/services/nfe/nfeSender.ts

---

Adicionar:

ts function detectarUfFromXml(xml: string): string | null {   const match = xml.match(/<emit>[\s\S]*?<UF>([A-Z]{2})<\/UF>/);   return match ? match[1] : null; } 

---

## 🔴 2. USAR UF DO XML

Localizar onde a UF é definida (ex: parâmetro da função):

---

### ❌ ANTES

ts const uf = "SP"; 

ou

ts enviarNFeSEFAZ(xml, "SP", ambiente); 

---

### ✅ DEPOIS

ts const ufXml = detectarUfFromXml(xmlAssinado); const ufFinal = ufXml ?? uf; 

---

## 🔴 3. USAR NO RESOLVE DA URL

ts const url = getSefazUrl(ufFinal, ambienteFinal); 

---

## 🔴 4. LOG DE SEGURANÇA (IMPORTANTE)

ts if (ufXml && ufXml !== uf) {   console.warn('[NFE_SEFAZ_UF_DIVERGENTE]', {     ufParametro: uf,     ufXml,     decisao: `usando UF do XML (${ufXml})`,   }); } 

---

## 🔴 5. FALLBACK OBRIGATÓRIO

Se XML falhar:

ts const ufFinal = ufXml ?? uf ?? "SP"; 

👉 garante compatibilidade total
