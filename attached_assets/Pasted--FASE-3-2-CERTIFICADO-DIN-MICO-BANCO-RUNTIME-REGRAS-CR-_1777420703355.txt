ts import { db } from '@/db'; import { eq } from 'drizzle-orm';  export async function getCompanyCertificate(companyId: number) {   return db.query.company_certificates.findFirst({     where: eq(company_certificates.company_id, companyId),   }); }  export async function saveCompanyCertificate(data: {   companyId: number;   certBase64: string;   certPassword: string; }) {   const existing = await getCompanyCertificate(data.companyId);    if (existing) {     return db       .update(company_certificates)       .set({         cert_base64: data.certBase64,         cert_password: data.certPassword,         updated_at: new Date(),       })       .where(eq(company_certificates.company_id, data.companyId));   }    return db.insert(company_certificates).values({     company_id: data.companyId,     cert_base64: data.certBase64,     cert_password: data.certPassword,   }); }
