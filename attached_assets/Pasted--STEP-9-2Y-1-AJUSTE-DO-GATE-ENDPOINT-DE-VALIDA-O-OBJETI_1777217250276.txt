### STEP 1 — AJUSTAR GUARD

Arquivo:

server/modules/nfe/faturamento.guard.ts

ts import { db } from "@/db"; import { sql } from "drizzle-orm";  export async function canEmitNFe(orderId: number) {   const result = await db.execute(sql`     SELECT        o.id,       o.status,       o.fiscal_status,       o.delivery_date,       c.contract_start_date,       c.contract_end_date     FROM orders o     JOIN companies c ON c.id = o.company_id     WHERE o.id = ${orderId}   `);    const order = result.rows?.[0];    if (!order) {     return { allowed: false, reason: "Pedido não encontrado" };   }    const now = new Date();    // Pedido cancelado   if (order.status === "CANCELLED") {     return { allowed: false, reason: "Pedido cancelado" };   }    // ✅ REGRA PRINCIPAL   if (order.fiscal_status !== "nota_liberada") {     return {       allowed: false,       reason: "Pedido ainda não liberado para faturamento"     };   }    // Entrega ainda não ocorreu   if (order.delivery_date && new Date(order.delivery_date) > now) {     return {       allowed: false,       reason: "Entrega ainda não realizada"     };   }    // Contrato ainda não iniciou   if (order.contract_start_date && new Date(order.contract_start_date) > now) {     return {       allowed: false,       reason: "Contrato ainda não iniciado"     };   }    // Contrato expirado   if (order.contract_end_date && new Date(order.contract_end_date) < now) {     return {       allowed: false,       reason: "Contrato da empresa expirado"     };   }    return { allowed: true }; }

ts import { canEmitNFe } from "@/modules/nfe/faturamento.guard";  app.get("/api/nfe/can-emit/:orderId", async (req, res) => {   try {     const orderId = Number(req.params.orderId);      if (!orderId) {       return res.status(400).json({         error: "orderId inválido"       });     }      const result = await canEmitNFe(orderId);      return res.json({       orderId,       ...result     });    } catch (error) {     console.error("[NFE_CAN_EMIT_ERROR]", error);      return res.status(500).json({       error: "Erro ao validar faturamento"     });   } });
