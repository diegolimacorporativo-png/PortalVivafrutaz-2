### STEP 1 — ENDPOINT DE EMISSÃO EM LOTE

ts app.post("/api/nfe/emitir-lote", async (req, res) => {   try {     const { orderIds } = req.body;      if (!Array.isArray(orderIds) || orderIds.length === 0) {       return res.status(400).json({         error: "Lista de pedidos inválida"       });     }      const results = await Promise.all(       orderIds.map(async (orderId: number) => {         try {           const check = await canEmitNFe(orderId);            if (!check.allowed) {             return {               orderId,               success: false,               reason: check.reason             };           }            const response = await fetch(             `${process.env.BASE_URL}/api/nfe/emitir`,             {               method: "POST",               headers: {                 "Content-Type": "application/json",                 Cookie: req.headers.cookie || ""               },               body: JSON.stringify({ orderId })             }           );            if (!response.ok) {             return {               orderId,               success: false,               reason: "Erro ao emitir NF"             };           }            return {             orderId,             success: true           };          } catch (err) {           return {             orderId,             success: false,             reason: "Erro interno"           };         }       })     );      console.log("[NFE_LOTE_EXECUTADO]", {       total: orderIds.length,       success: results.filter(r => r.success).length     });      return res.json({       total: results.length,       success: results.filter(r => r.success).length,       failed: results.filter(r => !r.success).length,       results     });    } catch (error) {     console.error("[NFE_LOTE_ERROR]", error);      return res.status(500).json({       error: "Erro ao emitir lote"     });   } });
