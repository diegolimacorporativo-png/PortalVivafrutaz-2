## STEP 9.3C — CRON INTELIGENTE DE FATURAMENTO

### OBJETIVO

Automatizar emissão de NF-e com base em:

- pedidos elegíveis
- regras do guard (canEmitNFe)
- ciclos de faturamento

SEM remover controle manual.

---

### 🚫 REGRAS

- NÃO alterar schema
- NÃO duplicar lógica
- NÃO emitir fora do guard
- NÃO rodar automático sem log

---

### STEP 1 — CRIAR JOB DE FATURAMENTO

Arquivo:
server/jobs/faturamento.cron.ts

ts import { db } from "@/db"; import { sql } from "drizzle-orm"; import { canEmitNFe } from "@/modules/nfe/faturamento.guard";  const CONCURRENCY_LIMIT = 5;  async function processInBatches<T>(   items: T[],   handler: (item: T) => Promise<any> ) {   const results: any[] = [];    for (let i = 0; i < items.length; i += CONCURRENCY_LIMIT) {     const batch = items.slice(i, i + CONCURRENCY_LIMIT);      const batchResults = await Promise.all(       batch.map((item) => handler(item))     );      results.push(...batchResults);   }    return results; }  export async function runFaturamentoCron() {   console.log("[CRON_FATURAMENTO_START]");    try {     const result = await db.execute(sql`       SELECT id       FROM orders       WHERE status != 'CANCELLED'         AND fiscal_status = 'nota_liberada'         AND delivery_date IS NOT NULL       LIMIT 500     `);      const orderIds = result.rows.map((r: any) => r.id);      const results = await processInBatches(orderIds, async (orderId: number) => {       try {         const check = await canEmitNFe(orderId);          if (!check.allowed) {           return { orderId, status: "blocked", reason: check.reason };         }          // chama o endpoint existente         const res = await fetch(           `${process.env.BASE_URL}/api/nfe/emitir`,           {             method: "POST",             headers: { "Content-Type": "application/json" },             body: JSON.stringify({ orderId })           }         );          if (!res.ok) {           return { orderId, status: "error" };         }          return { orderId, status: "success" };        } catch {         return { orderId, status: "error" };       }     });      console.log("[CRON_FATURAMENTO_RESULT]", {       total: results.length,       success: results.filter(r => r.status === "success").length,       blocked: results.filter(r => r.status === "blocked").length,       errors: results.filter(r => r.status === "error").length     });    } catch (error) {     console.error("[CRON_FATURAMENTO_ERROR]", error);   } }
