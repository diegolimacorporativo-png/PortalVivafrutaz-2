# ⚙️ FASE 3 — SERVICE (LEITURA PURA)

## Arquivo: server/services/alerts.preferences.ts (NOVO)

ts import { db } from "../db"; import { userNotificationPreferences } from "@shared/schema"; import { eq } from "drizzle-orm";  export async function getUserPreferences(userId: number) {   return db     .select()     .from(userNotificationPreferences)     .where(eq(userNotificationPreferences.userId, userId)); } 

---

## Função helper (futura integração)

ts export function isSeverityAllowed(   userMin: string,   alertSeverity: string ): boolean {   const order = ["INFO", "WARNING", "ALERT", "CRITICAL"];    return (     order.indexOf(alertSeverity) >= order.indexOf(userMin)   ); }

ts app.get(   "/api/admin/notifications/preferences",   requireAuthCore,   async (req, res) => {     const userId = req.user.id;      const data = await getUserPreferences(userId);      res.json(data);   } ); 

---

### POST — salvar/atualizar preferência

ts app.post(   "/api/admin/notifications/preferences",   requireAuthCore,   async (req, res) => {     const userId = req.user.id;     const { category, minSeverity, enabled } = req.body;      await db.insert(userNotificationPreferences).values({       userId,       category,       minSeverity,       enabled,     });      res.json({ ok: true });   } );
