ts async getReceivableBreakdown(id: number) {   const rows = await db     .select()     .from(financialTransactions)     .where(       and(         eq(financialTransactions.referenciaId, id),         eq(financialTransactions.referenciaTipo, "receivable"),         tenantWhere(financialTransactions)       )     );    let principal = 0;   let juros = 0;   let multa = 0;   let desconto = 0;    for (const row of rows) {     const valor = Number(row.valor);      if (row.descricao?.startsWith("[JUROS_CNAB]")) {       juros += valor;     } else if (row.descricao?.startsWith("[MULTA_CNAB]")) {       multa += valor;     } else if (row.descricao?.startsWith("[DESCONTO_CNAB]")) {       desconto += valor;     } else {       principal += valor;     }   }    return {     principal,     juros,     multa,     desconto,     totalRecebido: principal,     totalLiquido: principal + juros + multa - desconto,   }; }
