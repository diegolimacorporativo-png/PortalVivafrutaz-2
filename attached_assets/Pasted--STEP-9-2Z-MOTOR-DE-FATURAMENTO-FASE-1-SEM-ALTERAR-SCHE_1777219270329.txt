## STEP 1 — CRIAR ENGINE

Arquivo:

server/modules/nfe/faturamento.engine.ts

ts id="e1x93k" export function getFaturamentoContext(company: any, order: any) {   const now = new Date();    const context = {     tipo: "imediato",     prazoDias: 0,     podeEmitir: true,     motivo: "",   };    // 🔹 INTERPRETAÇÃO DO TIPO DE CLIENTE   switch (company.clientType) {     case "semanal":       context.tipo = "semanal";       break;     case "mensal":       context.tipo = "mensal";       break;     case "contratual":       context.tipo = "contratual";       break;     default:       context.tipo = "imediato";   }    // 🔹 EXTRAIR PRAZO (billingTerm = "30 dias", "45 dias", etc)   if (company.billingTerm) {     const match = company.billingTerm.match(/(\d+)/);     if (match) {       context.prazoDias = Number(match[1]);     }   }    // 🔹 BLOQUEIO POR CONTRATO   if (company.contractEndDate && new Date(company.contractEndDate) < now) {     context.podeEmitir = false;     context.motivo = "Contrato expirado";     return context;   }    // 🔹 REGRAS POR TIPO    if (context.tipo === "semanal") {     const delivery = new Date(order.deliveryDate);     const diff = Math.floor(       (now.getTime() - delivery.getTime()) / (1000 * 60 * 60 * 24)     );      if (diff < 7) {       context.podeEmitir = false;       context.motivo = "Aguardando fechamento semanal";     }   }    if (context.tipo === "mensal") {     const delivery = new Date(order.deliveryDate);      if (       delivery.getMonth() === now.getMonth() &&       delivery.getFullYear() === now.getFullYear()     ) {       context.podeEmitir = false;       context.motivo = "Faturamento apenas no mês seguinte";     }   }    return context; }
